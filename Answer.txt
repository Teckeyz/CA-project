#Short explanation of the code, line-by-line
#include <stdio.h>

This line includes the standard input/output library which allows us to use functions like printf and scanf.

#include <math.h>

This line includes the math library which provides us with mathematical functions like pow, sqrt, and log.

double add(double a, double b) {
    return a + b;
}

This function takes two double values as arguments and returns their sum.

double subtract(double a, double b) {
    return a - b;
}

This function returns the difference between two double values.

double multiply(double a, double b) {
    return a * b;
}

This function returns the product of two double values.

double divide(double a, double b) {
    if(b != 0) return a / b;
    else return 0; // Error: Division by zero
}

This function returns the quotient of two double values. It checks for division by zero and returns 0 in that case to avoid an error.

double power(double a, double b) {
    return pow(a, b);
}

This function uses the pow function from math.h to calculate and return a raised to the power of b.

double squareRoot(double a) {
    if(a >= 0) return sqrt(a);
    else return 0; // Error: Negative number
}

This function calculates the square root of a double value using sqrt from math.h. It checks if the number is non-negative before attempting to find the square root.

double logarithm(double a) {
    if(a > 0) return log(a);
    else return 0; // Error: Non-positive number
}

This function calculates the natural logarithm of a double value using log from math.h. It ensures the number is positive before computing the logarithm.

int main() {

This is the main function where the program execution begins.

    double num1, num2, result;
    int choice;

Here, we declare variables to store two numbers (num1 and num2), the result of the operation, and the user's choice of operation.

    printf("Advanced Calculator\n");

This line prints the title of the program to the console.

    printf("1. Addition\n");
    printf("2. Subtraction\n");
    printf("3. Multiplication\n");
    printf("4. Division\n");
    printf("5. Power\n");
    printf("6. Square Root\n");
    printf("7. Logarithm\n");
    printf("Enter your choice: ");
    scanf("%d", &choice);

These lines display the menu of operations and read the user's choice into the choice variable.

    if(choice == 6 || choice == 7) {
        printf("Enter number: ");
        scanf("%lf", &num1);
    } else {
        printf("Enter first number: ");
        scanf("%lf", &num1);
        printf("Enter second number: ");
        scanf("%lf", &num2);
    }

This conditional block asks for one or two numbers from the user, depending on the operation chosen.

    switch(choice) {

This switch statement will execute different blocks of code based on the user's choice.

        case 1:
            result = add(num1, num2);
            break;

If the user chooses 1, the add function is called and the result is stored.

        // Similar cases for 2, 3, 4, 5...

Similar blocks of code are used for subtraction, multiplication, division, and power.

        case 6:
            result = squareRoot(num1);
            break;
        case 7:
            result = logarithm(num1);
            break;

For choices 6 and 7, the squareRoot and logarithm functions are called respectively.

        default:
            printf("Invalid choice!\n");
            return 0;
    }

If the user enters an invalid choice, an error message is printed and the program exits.

    printf("Result: %lf\n", result);
    return 0;
}
